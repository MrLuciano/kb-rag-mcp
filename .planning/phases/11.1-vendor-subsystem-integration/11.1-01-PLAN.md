**Type:** Remediation
**Priority:** Critical

## Goal

2. Adding vendor/subsystem filter parameters

### 10. Add integration tests

**New test:** `test_search_with_vendor_filter()`
```python
@pytest.mark.asyncio
async def test_search_with_vendor_filter(vector_store, sample_chunks):
    """Test search with vendor filter."""
    # Upsert chunks with vendor=OpenText
    chunks_with_vendor = [
        {**chunk, "vendor": "OpenText"}
        for chunk in sample_chunks
    ]
    await vector_store.upsert_chunks(chunks_with_vendor)
    
    # Search with vendor filter
    query_vector = [0.1] * 384
    results = await vector_store.search(
        query_vector=query_vector,
        top_k=5,
        vendor="OpenText",
    )
    
    # Verify all results have vendor=OpenText
    assert len(results) > 0
    for result in results:
        assert result["vendor"] == "OpenText"
```

async def test_search_result_includes_vendor_subsystem(vector_store, sample_chunks):
    """Test search results include vendor and subsystem fields."""
    # Upsert chunks with vendor and subsystem
    chunks_with_metadata = [
        {**chunk, "vendor": "OpenText", "subsystem": "Archive"}
        for chunk in sample_chunks
    ]
    await vector_store.upsert_chunks(chunks_with_metadata)
    
    # Search without filters
    query_vector = [0.1] * 384
    results = await vector_store.search(query_vector=query_vector, top_k=5)
    
    # Verify result dict includes vendor and subsystem
    assert len(results) > 0
    result = results[0]
    assert "vendor" in result
    assert "subsystem" in result
    assert result["vendor"] == "OpenText"
    assert result["subsystem"] == "Archive"

@pytest.mark.asyncio
async def test_search_kb_with_vendor_filter(mock_vector_store, mock_embed_client):
    """Test search_kb MCP tool with vendor filter."""
    # Mock search results with vendor
    mock_vector_store.search.return_value = [
        {
            "id": "test1",
            "text": "Test content",
            "score": 0.95,
            "vendor": "OpenText",
            "product": "WebReports",
            "subsystem": "Archive",
            "source": "test.pdf",
        }
    ]
    
    # Call search_kb with vendor filter
    result = await _search_kb(
        arguments={"query": "test query", "vendor": "OpenText"},
        store=mock_vector_store,
        embed_client=mock_embed_client,
    )
    
    # Verify vendor passed to vector_store.search()
    mock_vector_store.search.assert_called_once()
    call_kwargs = mock_vector_store.search.call_args.kwargs
    assert call_kwargs["vendor"] == "OpenText"
    
    # Verify result includes vendor
    assert "vendor" in result[0].text
    assert "OpenText" in result[0].text
