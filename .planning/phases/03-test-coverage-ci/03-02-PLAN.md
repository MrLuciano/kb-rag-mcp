```python
import pytest
from unittest.mock import AsyncMock, MagicMock, patch
import os

@pytest.fixture(autouse=True)
def set_openai_compat_backend(monkeypatch):
    monkeypatch.setenv("EMBED_BACKEND", "openai-compat")

def _mock_openai_response(dims: int = 768):
    mock_resp = MagicMock()
    mock_resp.status_code = 200
    mock_resp.raise_for_status = MagicMock()
    mock_resp.json.return_value = {"data": [{"embedding": [0.1] * dims}]}
    return mock_resp
```

For async httpx mock:
```python
@pytest.mark.asyncio
async def test_get_embedding_openai_compat():
    mock_client = AsyncMock()
    mock_client.post.return_value = _mock_openai_response(768)
    
    with patch("httpx.AsyncClient") as MockClient:
        MockClient.return_value.__aenter__.return_value = mock_client
        MockClient.return_value.__aexit__.return_value = None
        
        # Also disable cache to ensure httpx is called
        import kb_server.embed_client as ec
        with patch.object(ec, "_cache_manager", None):
            result = await ec.get_embedding("test text", use_cache=False)
    
    assert isinstance(result, list)
    assert len(result) == 768
```
