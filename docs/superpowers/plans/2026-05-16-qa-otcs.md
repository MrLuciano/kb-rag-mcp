# QA Evaluation — End-to-End Pipeline Validation Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a self-contained QA runner (`qa/run_qa.py`) that ingests the full evaluation corpus (271 docs), runs 20 curated queries, computes heuristic RAGAS-lite metrics, and writes `docs/QA_REPORT.md`.

**Architecture:** A single Python orchestrator in `qa/` calls the existing `ingest.run_ingest()` function after starting Qdrant via Docker. Queries run directly against `VectorStore.search()`. Metrics are pure functions in `qa/metrics.py`. Report rendering is a pure function in `qa/report.py`. The golden dataset is updated in-place after the probe. No production files are modified.

**Tech Stack:** Python 3.11, `fastembed` (dense backend `BAAI/bge-small-en-v1.5`, 384-dim), `qdrant-client`, `docker` CLI (subprocess), existing `ingest/ingest.py` and `server/vector_store.py`.

**Key constraint:** `embed_client.py` has no fastembed dense backend. The QA runner adds a self-contained `qa/embedder.py` that wraps `fastembed.TextEmbedding` directly — no modifications to production `embed_client.py`. The collection is named `qa_otcs` to avoid colliding with any existing `kb_docs` collection.

```python
"""Pure metric functions for QA evaluation. No I/O, no side effects."""
import statistics
from typing import Any


# ── Pass thresholds ───────────────────────────────────────────────────────────

THRESHOLDS = {
    "hit_rate":         {"min": 0.70, "direction": "ge"},
    "mrr":              {"min": 0.50, "direction": "ge"},
    "zero_result_rate": {"max": 0.10, "direction": "le"},
    "score_p50":        {"min": 0.45, "direction": "ge"},
    "score_p95":        {"min": 0.65, "direction": "ge"},
    "latency_p50_ms":   {"max": 500,  "direction": "le"},
    "latency_p95_ms":   {"max": 2000, "direction": "le"},
}


def compute_hit_rate(results: list[dict]) -> float:
    """Fraction of queries where at least one result matched expected keywords."""
    if not results:
        return 0.0
    return sum(1 for r in results if r.get("hit", False)) / len(results)


def compute_mrr(results: list[dict]) -> float:
    """Mean Reciprocal Rank @5. rank=0 means no hit."""
    if not results:
        return 0.0
    rr_sum = sum(1.0 / r["rank"] if r.get("rank", 0) > 0 else 0.0 for r in results)
    return rr_sum / len(results)


def compute_score_stats(results: list[dict]) -> dict[str, float]:
    """Percentile and mean stats over top_score values."""
    scores = [r.get("top_score", 0.0) for r in results]
    if not scores:
        return {"p25": 0.0, "p50": 0.0, "p75": 0.0, "p95": 0.0, "mean": 0.0}
    scores_sorted = sorted(scores)
    n = len(scores_sorted)

    def pct(p: float) -> float:
        idx = (p / 100) * (n - 1)
        lo, hi = int(idx), min(int(idx) + 1, n - 1)
        return scores_sorted[lo] + (scores_sorted[hi] - scores_sorted[lo]) * (idx - lo)

    return {
        "p25": round(pct(25), 4),
        "p50": round(pct(50), 4),
        "p75": round(pct(75), 4),
        "p95": round(pct(95), 4),
        "mean": round(statistics.mean(scores), 4),
    }


def compute_latency_stats(results: list[dict]) -> dict[str, float]:
    """Percentile and mean stats over latency_ms values."""
    latencies = [r.get("latency_ms", 0.0) for r in results]
    if not latencies:
        return {"p50": 0.0, "p95": 0.0, "mean": 0.0}
    latencies_sorted = sorted(latencies)
    n = len(latencies_sorted)

    def pct(p: float) -> float:
        idx = (p / 100) * (n - 1)
        lo, hi = int(idx), min(int(idx) + 1, n - 1)
        return latencies_sorted[lo] + (latencies_sorted[hi] - latencies_sorted[lo]) * (idx - lo)

    return {
        "p50": round(pct(50), 1),
        "p95": round(pct(95), 1),
        "mean": round(statistics.mean(latencies), 1),
    }


def compute_zero_result_rate(results: list[dict]) -> float:
    """Fraction of queries returning zero results (top_score == 0.0)."""
    if not results:
        return 0.0
    return sum(1 for r in results if r.get("top_score", 0.0) == 0.0) / len(results)


def compute_verdict(metrics: dict) -> tuple[str, list[str]]:
    """
    Returns ("PASS"|"FAIL", [reason strings]).
    Checks: hit_rate, mrr, zero_result_rate, score_stats.p50.
    """
    reasons = []
    if metrics.get("hit_rate", 0.0) < THRESHOLDS["hit_rate"]["min"]:
        reasons.append(
            f"hit_rate={metrics['hit_rate']:.2f} < {THRESHOLDS['hit_rate']['min']}"
        )
    if metrics.get("mrr", 0.0) < THRESHOLDS["mrr"]["min"]:
        reasons.append(
            f"mrr={metrics['mrr']:.2f} < {THRESHOLDS['mrr']['min']}"
        )
    if metrics.get("zero_result_rate", 0.0) > THRESHOLDS["zero_result_rate"]["max"]:
        reasons.append(
            f"zero_result_rate={metrics['zero_result_rate']:.2f} > {THRESHOLDS['zero_result_rate']['max']}"
        )
    p50 = metrics.get("score_stats", {}).get("p50", 0.0)
    if p50 < THRESHOLDS["score_p50"]["min"]:
        reasons.append(
            f"score_p50={p50:.2f} < {THRESHOLDS['score_p50']['min']}"
        )
    return ("FAIL" if reasons else "PASS"), reasons
```

```python
"""Validate queries.json schema — every entry must be well-formed."""
import json
import os
import pytest

QUERIES_PATH = os.path.join(os.path.dirname(__file__), "..", "qa", "queries.json")

REQUIRED_FIELDS = {"id", "category", "query", "expected_doc_keywords", "min_expected_score"}
VALID_CATEGORIES = {
    "installation", "administration", "search_index",
    "workflow_forms", "sdk_development", "webreports", "records_management",
}


def load_queries():
    with open(QUERIES_PATH) as f:
        return json.load(f)


def test_queries_file_exists():
    assert os.path.exists(QUERIES_PATH), f"queries.json not found at {QUERIES_PATH}"


def test_queries_count():
    queries = load_queries()
    assert len(queries) == 20, f"Expected 20 queries, got {len(queries)}"


def test_queries_required_fields():
    queries = load_queries()
    for q in queries:
        missing = REQUIRED_FIELDS - set(q.keys())
        assert not missing, f"Query {q.get('id', '?')} missing fields: {missing}"


def test_queries_unique_ids():
    queries = load_queries()
    ids = [q["id"] for q in queries]
    assert len(ids) == len(set(ids)), "Duplicate query IDs found"


def test_queries_valid_categories():
    queries = load_queries()
    for q in queries:
        assert q["category"] in VALID_CATEGORIES, (
            f"Query {q['id']} has unknown category: {q['category']}"
        )


def test_queries_keywords_non_empty():
    queries = load_queries()
    for q in queries:
        assert isinstance(q["expected_doc_keywords"], list), (
            f"Query {q['id']}: expected_doc_keywords must be a list"
        )
        assert len(q["expected_doc_keywords"]) >= 1, (
            f"Query {q['id']}: expected_doc_keywords must not be empty"
        )


def test_queries_score_threshold_valid():
    queries = load_queries()
    for q in queries:
        assert 0.0 <= q["min_expected_score"] <= 1.0, (
            f"Query {q['id']}: min_expected_score out of range"
        )


def test_queries_category_distribution():
    queries = load_queries()
    from collections import Counter
    counts = Counter(q["category"] for q in queries)
    # Each category must appear at least once
    for cat in VALID_CATEGORIES:
        assert counts[cat] >= 1, f"Category '{cat}' has no queries"
```

```python
from datetime import datetime


def render_report(metrics: dict) -> str:
    """
    Render a QA report as a Markdown string.

    Args:
        metrics: dict produced by run_qa.py containing all metric values,
                 query_results list, corpus metadata, and verdict.

    Returns:
        Markdown string ready to write to docs/QA_REPORT.md.
    """
    now = datetime.utcnow().strftime("%Y-%m-%d %H:%M UTC")
    verdict = metrics.get("verdict", "UNKNOWN")
    reasons = metrics.get("verdict_reasons", [])
    ss = metrics.get("score_stats", {})
    ls = metrics.get("latency_stats", {})

    t = THRESHOLDS = {
        "hit_rate": 0.70, "mrr": 0.50, "zero_result_rate": 0.10,
        "score_p50": 0.45, "score_p95": 0.65,
        "latency_p50": 500, "latency_p95": 2000,
    }

    def status(val, threshold, direction="ge"):
        if direction == "ge":
            return "✅" if val >= threshold else "❌"
        return "✅" if val <= threshold else "❌"

    lines = [
        "# KB-RAG-MCP QA Report — Evaluation Corpus",

        f"**Collection:** `{metrics.get('collection', 'N/A')}`  ",
        f"**Total chunks indexed:** {metrics.get('total_chunks', '?')}  ",
        f"**Ingest duration:** {metrics.get('ingest_duration_s', 0):.0f}s  ",
        "",
        "---",
        "",
        f"## Overall Verdict: {verdict}",
        "",
    ]

    if verdict == "FAIL":
        lines.append("**Failure reasons:**")
        for r in reasons:
            lines.append(f"- {r}")
        lines.append("")
    else:
        lines.append("All primary metrics met thresholds.")
        lines.append("")

    lines += [
        "---",
        "",
        "## Aggregate Metrics",
        "",
        "| Metric | Value | Threshold | Status |",
        "|---|---|---|---|",
        f"| Hit Rate @5 | {metrics.get('hit_rate', 0):.1%} | ≥ 70% | {status(metrics.get('hit_rate', 0), t['hit_rate'])} |",
        f"| MRR @5 | {metrics.get('mrr', 0):.3f} | ≥ 0.50 | {status(metrics.get('mrr', 0), t['mrr'])} |",

        f"| Score p50 | {ss.get('p50', 0):.3f} | ≥ 0.45 | {status(ss.get('p50', 0), t['score_p50'])} |",

        f"| Latency p95 | {ls.get('p95', 0):.0f} ms | ≤ 2000 ms | {status(ls.get('p95', 0), t['latency_p95'], 'le')} |",
        f"| Score p25 | {ss.get('p25', 0):.3f} | — | — |",
        f"| Score p75 | {ss.get('p75', 0):.3f} | — | — |",
        f"| Score mean | {ss.get('mean', 0):.3f} | — | — |",
        f"| Latency mean | {ls.get('mean', 0):.0f} ms | — | — |",
        "",
        "---",
        "",
        "## Per-Query Results",
        "",
        "| ID | Category | Query | Hit | Rank | Score | Latency ms | Top Source |",
        "|---|---|---|---|---|---|---|---|",
    ]

    query_results = metrics.get("query_results", [])
    zero_result_ids = []

    for r in query_results:
        hit_icon = "✅" if r.get("hit") else "❌"
        rank = r.get("rank", 0) or "—"
        score = f"{r.get('top_score', 0.0):.3f}"
        latency = f"{r.get('latency_ms', 0.0):.0f}"
        sources = r.get("top_sources", [])
        top_source = sources[0] if sources else "—"
        # truncate long filenames
        if len(top_source) > 50:
            top_source = top_source[:47] + "..."
        query_trunc = r.get("query", "")[:55]
        lines.append(
            f"| {r.get('id', '?')} | {r.get('category', '?')} "
            f"| {query_trunc} | {hit_icon} | {rank} "
            f"| {score} | {latency} | {top_source} |"
        )
        if r.get("top_score", 0.0) == 0.0:
            zero_result_ids.append(r.get("id", "?"))

    lines.append("")

    if zero_result_ids:
        lines += [
            "---",
            "",
            "## Zero-Result Queries",
            "",
            "These queries returned no results. Consider reviewing chunking strategy,",
            "ensuring the relevant documents were ingested, or adjusting query phrasing.",

            qr = next((r for r in query_results if r.get("id") == qid), {})

    lines += [
        "---",
        "",
        "## Recommendations",
        "",
    ]

    if verdict == "PASS":

    else:

        if metrics.get("zero_result_rate", 0.0) > t["zero_result_rate"]:

        if ss.get("p50", 1.0) < t["score_p50"]:
            lines.append("- **Low scores:** Consider a larger embedding model or hybrid search to improve semantic matching.")

    lines.append("")
    lines.append("---")
    lines.append("*Report generated by `qa/run_qa.py`. Re-run with `--skip-ingest` to refresh queries without re-ingesting.*")
    lines.append("")

    return "\n".join(lines)
```

```python
"""
Self-contained FastEmbed dense embedder for the QA runner.

Uses BAAI/bge-small-en-v1.5 (384-dim, ~60 MB).
Model is lazy-loaded on first call to embed().

This module is intentionally separate from server/embed_client.py
to avoid adding a fastembed dense backend to production code.
"""
import asyncio
import logging
from typing import Optional

log = logging.getLogger(__name__)

DEFAULT_MODEL = "BAAI/bge-small-en-v1.5"
DEFAULT_DIM = 384


class QAEmbedder:
    """Wraps fastembed.TextEmbedding for synchronous and async use."""

    def __init__(self, model_name: str = DEFAULT_MODEL):
        self.model_name = model_name
        self._model = None  # lazy loaded

    @property
    def dim(self) -> int:
        """Embedding dimension for the configured model."""
        dims = {
            "BAAI/bge-small-en-v1.5": 384,
            "BAAI/bge-base-en-v1.5": 768,
            "BAAI/bge-large-en-v1.5": 1024,
        }
        return dims.get(self.model_name, DEFAULT_DIM)

    def _load_model(self):
        if self._model is None:
            log.info(f"Loading FastEmbed model: {self.model_name}")
            from fastembed import TextEmbedding
            self._model = TextEmbedding(model_name=self.model_name)
            log.info(f"FastEmbed model loaded (dim={self.dim})")

    def embed(self, text: str) -> list[float]:
        """Embed a single text string. Loads model on first call."""
        self._load_model()
        vectors = list(self._model.embed([text]))
        return vectors[0].tolist()

    def embed_batch(self, texts: list[str]) -> list[list[float]]:
        """Embed a batch of texts. More efficient than calling embed() in a loop."""
        if not texts:
            return []
        self._load_model()
        vectors = list(self._model.embed(texts))
        return [v.tolist() for v in vectors]

    async def aembed(self, text: str) -> list[float]:
        """Async shim — runs embed() in a thread executor."""
        loop = asyncio.get_event_loop()
        return await loop.run_in_executor(None, self.embed, text)

    async def aembed_batch(self, texts: list[str]) -> list[list[float]]:
        """Async shim — runs embed_batch() in a thread executor."""
        loop = asyncio.get_event_loop()
        return await loop.run_in_executor(None, self.embed_batch, texts)
```

```python
#!/usr/bin/env python3
"""
QA Runner — End-to-end pipeline validation with evaluation corpus.

Usage:
    python qa/run_qa.py [--otcs-path PATH] [--workers N] [--skip-ingest]

Environment:
    CORPUS_PATH   Path to document corpus (default: /mnt/c/Users/luciano.marinho/git/corpus)
    QA_WORKERS  Ingest worker count (default: 4)
"""

import argparse
import asyncio
import json
import logging
import os
import subprocess
import sys
import time
from pathlib import Path
from typing import Optional

# ── Paths ─────────────────────────────────────────────────────────────────────

REPO_ROOT = Path(__file__).parent.parent
SERVER_DIR = REPO_ROOT / "server"
INGEST_DIR = REPO_ROOT / "ingest"

sys.path.insert(0, str(SERVER_DIR))
sys.path.insert(0, str(INGEST_DIR))

QUERIES_PATH = Path(__file__).parent / "queries.json"
DEFAULT_REPORT_PATH = REPO_ROOT / "docs" / "QA_REPORT.md"
DEFAULT_GOLDEN_PATH = REPO_ROOT / "server" / "evaluation" / "golden_dataset.json"
DEFAULT_CORPUS_PATH = Path(
    os.getenv("CORPUS_PATH", "/mnt/c/Users/luciano.marinho/git/corpus")
)
DEFAULT_COLLECTION = "qa_eval"

log = logging.getLogger("qa_runner")
logging.basicConfig(level=logging.INFO, format="%(asctime)s [%(levelname)s] %(message)s")


# ── QARunner ─────────────────────────────────────────────────────────────────

class QARunner:
    """Orchestrates the full QA pipeline."""

    def __init__(
        self,
        corpus_path: Path = DEFAULT_CORPUS_PATH,
        collection: str = DEFAULT_COLLECTION,
        workers: int = 4,
        report_path: Path = DEFAULT_REPORT_PATH,
        golden_dataset_path: Path = DEFAULT_GOLDEN_PATH,
        queries_override: Optional[list] = None,
        skip_ingest: bool = False,
    ):
        self.corpus_path = Path(corpus_path)
        self.collection = collection
        self.workers = workers
        self.report_path = Path(report_path)
        self.golden_dataset_path = Path(golden_dataset_path)
        self.queries = queries_override or self._load_queries()
        self.skip_ingest = skip_ingest
        self._qdrant_container_id: Optional[str] = None

    def _load_queries(self) -> list:
        with open(QUERIES_PATH) as f:
            return json.load(f)

    # ── Stage 1: Preflight ────────────────────────────────────────────────────

    def _preflight(self):
        log.info("=== Stage 1: Preflight ===")
        assert self.corpus_path.exists(), f"Corpus path not found: {self.corpus_path}"
        result = subprocess.run(["docker", "info"], capture_output=True)
        assert result.returncode == 0, "Docker is not running or not available"
        doc_count = sum(
            1 for _ in self.corpus_path.rglob("*")
            if _.is_file() and _.suffix.lower() in
            {".pdf", ".docx", ".xlsx", ".pptx", ".txt", ".md"}
        )
        log.info(f"Corpus: {self.corpus_path} ({doc_count} documents)")
        return doc_count

    # ── Stage 2: Qdrant ───────────────────────────────────────────────────────

    def _start_qdrant(self):
        log.info("=== Stage 2: Starting Qdrant ===")
        result = subprocess.run(
            ["docker", "run", "-d", "--rm", "-p", "6333:6333",
             "--name", "qa_qdrant", "qdrant/qdrant"],
            capture_output=True, text=True,
        )
        if result.returncode != 0:
            # Already running?
            if "already in use" in result.stderr or "Conflict" in result.stderr:
                log.warning("Qdrant container already running — using existing instance")
            else:
                raise RuntimeError(f"Failed to start Qdrant: {result.stderr}")
        else:
            self._qdrant_container_id = result.stdout.strip()
            log.info(f"Qdrant container started: {self._qdrant_container_id[:12]}")

        # Wait for Qdrant to be ready
        import httpx
        deadline = time.time() + 60
        while time.time() < deadline:
            try:
                r = httpx.get("http://localhost:6333/healthz", timeout=3)
                if r.status_code == 200:
                    log.info("Qdrant is ready")
                    return
            except Exception:
                pass
            time.sleep(2)
        raise RuntimeError("Qdrant did not become ready within 60s")

    def _stop_qdrant(self):
        if self._qdrant_container_id:
            subprocess.run(
                ["docker", "stop", "qa_qdrant"],
                capture_output=True,
            )
            log.info("Qdrant container stopped")

    # ── Stage 3: Ingest ───────────────────────────────────────────────────────

    async def _ingest(self) -> tuple[int, float]:
        """Run ingest. Returns (total_chunks, duration_seconds)."""
        log.info(f"=== Stage 3: Ingest (workers={self.workers}) ===")

        # Set env vars for this process so ingest + embed_client pick them up
        os.environ["QDRANT_COLLECTION"] = self.collection
        os.environ["EMBED_BACKEND"] = "qa_fastembed"  # injected below

        # Inject qa_fastembed backend into embed_client at runtime
        from qa.embedder import QAEmbedder
        embedder = QAEmbedder()

        import embed_client as ec
        # Patch get_embedding and get_embeddings_batch to use our embedder
        async def _get_embedding(text: str, use_cache: bool = True) -> list[float]:
            return await embedder.aembed(text)

        async def _get_embeddings_batch(
            texts: list[str], batch_size: int = 32, use_cache: bool = True
        ) -> list[list[float]]:
            return await embedder.aembed_batch(texts)

        def _get_embed_dim() -> int:
            return embedder.dim

        ec.get_embedding = _get_embedding
        ec.get_embeddings_batch = _get_embeddings_batch
        ec.get_embed_dim = _get_embed_dim
        ec.MODEL = embedder.model_name
        ec.BACKEND = "fastembed"

        from ingest import run_ingest

        t0 = time.time()
        await run_ingest(
            docs_path=self.corpus_path,
            product="otcs",
            workers=self.workers,
            clean=True,
            force=True,
        )
        duration = time.time() - t0

        # Count chunks in collection
        from vector_store import VectorStore
        store = VectorStore()
        os.environ["QDRANT_COLLECTION"] = self.collection
        await store.connect()
        info = await store.client.get_collection(self.collection)
        total_chunks = info.points_count or 0
        await store.client.close()

        log.info(f"Ingest complete: {total_chunks} chunks in {duration:.1f}s")
        return total_chunks, duration

    # ── Stage 4: Probe ────────────────────────────────────────────────────────

    async def _probe(self, embedder) -> list[dict]:
        """Run all queries against the store. Returns per-query result dicts."""
        log.info("=== Stage 4: Probe ===")

        from vector_store import VectorStore
        os.environ["QDRANT_COLLECTION"] = self.collection
        store = VectorStore()
        await store.connect()

        results = []
        for q in self.queries:
            t0 = time.time()
            vector = await embedder.aembed(q["query"])
            search_results = await store.search(
                vector=vector,
                top_k=5,
                filter_type=None,
                product=None,
                doc_type=None,
                version=None,
            )
            latency_ms = (time.time() - t0) * 1000

            # Determine hit: does any result's source_file contain a keyword?
            hit = False
            rank = 0
            top_score = 0.0
            top_sources = []

            if search_results:
                top_score = search_results[0].get("score", 0.0)
                top_sources = [
                    Path(r.get("source_file", "")).name
                    for r in search_results
                ]
                keywords = [kw.lower() for kw in q.get("expected_doc_keywords", [])]
                for i, r in enumerate(search_results, 1):
                    src = r.get("source_file", "").lower()
                    if any(kw in src for kw in keywords):
                        hit = True
                        rank = i
                        break

            result = {
                "id": q["id"],
                "category": q["category"],
                "query": q["query"],
                "hit": hit,
                "rank": rank,
                "top_score": top_score,
                "latency_ms": latency_ms,
                "top_sources": top_sources[:5],
                "retrieved_contexts": [
                    r.get("text", "")[:800] for r in search_results[:5]
                ],
            }
            results.append(result)
            log.info(
                f"  [{q['id']}] hit={hit} rank={rank} "
                f"score={top_score:.3f} latency={latency_ms:.0f}ms"
            )

        await store.client.close()
        return results

    # ── Stage 5: Metrics ─────────────────────────────────────────────────────

    def _compute_metrics(self, query_results: list[dict], doc_count: int,
                         total_chunks: int, ingest_duration: float,
                         embedder) -> dict:
        log.info("=== Stage 5: Metrics ===")
        from qa.metrics import (
            compute_hit_rate, compute_mrr, compute_score_stats,
            compute_latency_stats, compute_zero_result_rate, compute_verdict,
        )

        hit_rate = compute_hit_rate(query_results)
        mrr = compute_mrr(query_results)
        score_stats = compute_score_stats(query_results)
        latency_stats = compute_latency_stats(query_results)
        zero_result_rate = compute_zero_result_rate(query_results)

        metrics = {
            "corpus_path": str(self.corpus_path),
            "corpus_docs": doc_count,
            "embed_model": embedder.model_name,
            "embed_dim": embedder.dim,
            "collection": self.collection,
            "ingest_duration_s": ingest_duration,
            "total_chunks": total_chunks,
            "hit_rate": hit_rate,
            "mrr": mrr,
            "zero_result_rate": zero_result_rate,
            "score_stats": score_stats,
            "latency_stats": latency_stats,
            "query_results": query_results,
        }
        verdict, reasons = compute_verdict(metrics)
        metrics["verdict"] = verdict
        metrics["verdict_reasons"] = reasons
        return metrics

    # ── Stage 6: Report ───────────────────────────────────────────────────────

    def _write_report(self, metrics: dict):
        log.info(f"=== Stage 6: Report → {self.report_path} ===")
        from qa.report import render_report
        self.report_path.parent.mkdir(parents=True, exist_ok=True)
        content = render_report(metrics)
        self.report_path.write_text(content, encoding="utf-8")
        log.info(f"Report written: {self.report_path}")

    # ── Stage 7: Golden dataset ───────────────────────────────────────────────

    def _update_golden_dataset(self, metrics: dict):
        log.info(f"=== Stage 7: Golden dataset → {self.golden_dataset_path} ===")
        golden = []
        for r in metrics.get("query_results", []):
            entry = {
                "query": r["query"],
                "expected_answer": r["retrieved_contexts"][0][:400]
                    if r.get("retrieved_contexts") else "",
                "expected_docs": [r["top_sources"][0]] if r.get("top_sources") else [],
                "retrieved_contexts": r.get("retrieved_contexts", []),
                "metadata": {
                    "product": "otcs",
                    "category": r.get("category", ""),
                    "hit": r.get("hit", False),
                    "top_score": r.get("top_score", 0.0),
                },
            }
            golden.append(entry)

        self.golden_dataset_path.parent.mkdir(parents=True, exist_ok=True)
        with open(self.golden_dataset_path, "w", encoding="utf-8") as f:
            json.dump(golden, f, indent=2, ensure_ascii=False)
        log.info(f"Golden dataset updated: {len(golden)} entries")

    # ── Main ─────────────────────────────────────────────────────────────────

    async def run(self) -> dict:
        """Execute all QA stages. Returns metrics dict."""
        from qa.embedder import QAEmbedder
        embedder = QAEmbedder()

        doc_count = self._preflight()
        self._start_qdrant()

        try:
            if self.skip_ingest:
                log.info("Skipping ingest (--skip-ingest)")
                total_chunks, ingest_duration = 0, 0.0
            else:
                total_chunks, ingest_duration = await self._ingest()

            query_results = await self._probe(embedder)
            metrics = self._compute_metrics(
                query_results, doc_count, total_chunks, ingest_duration, embedder
            )
            self._write_report(metrics)
            self._update_golden_dataset(metrics)

            log.info(f"\n{'='*50}")
            log.info(f"QA VERDICT: {metrics['verdict']}")
            if metrics["verdict_reasons"]:
                for r in metrics["verdict_reasons"]:
                    log.info(f"  ✗ {r}")
            log.info(f"Hit Rate: {metrics['hit_rate']:.1%}  MRR: {metrics['mrr']:.3f}")
            log.info(f"Score p50: {metrics['score_stats']['p50']:.3f}  "
                     f"Zero-result: {metrics['zero_result_rate']:.1%}")
            log.info(f"Report: {self.report_path}")
            log.info(f"{'='*50}\n")

            return metrics

        finally:
            self._stop_qdrant()
```
